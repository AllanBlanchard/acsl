  term ::= "\at" "(" term "," label-id ")"
  \ 
  pred ::= "\at" "(" pred "," label-id ")" 
  \

             | "LoopEntry" | "LoopCurrent" | "Init" | ;
             | id [An id must be a C program label in scope]
\end{syntax}
