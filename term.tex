\begin{syntax}
  term ::= id | uop term | term bop term ;  
       | term "->" id | term "." id  ;
       | "\let" id "=" term ";" term ;
       | "(" \tau ")" term  ; cast 
       | term "[" term "]" ;
       | id "(" term ("," term)* ")" ; function application
       | "(" term ")" ;
       | "sizeof" "(" term ")" ;
       | "sizeof" "(" \tau ")" ;
       | "offsetof" "(" ... ")" ;
       | "alignof" "(" \tau ")" ;
       | term "?" term ":" term ;
       | "\old" "(" term ")" | "\at" "(" term "," label ")" ;
       | "\result" ;
       | "(" term ("," term)+ ")" ; tuples
       | "\let" "(" id (, id)+ ")" "=" term ";" term ; \experimental
       \
  bop ::= "+" | "-" | "*" | "/" | "%" ;
       | "==" | "!=" | "<=" | ">=" | ">" | "<" ;
       | "&" | "|" | "=>" | "<=>" ; bitwise operations 
       \
  uop ::= "-" | "*" | "&" \
  rop ::= "==" | "!=" | "<=" | ">=" | ">" | "<" \
  pred ::= "\true" | "\false" ;
       | term (rop term)* ;
       | "\let" id "=" term ";" pred ;
       | id "(" term ("," term)* ")" ; predicate application
       | "(" pred ")" ;
       | pred "&&" pred | pred "||" pred ;
       | pred "==>" pred | pred "<==>" pred ;
       | "!" pred ;
       | pred "?" pred ":" pred ;
       | "\forall" binders ";" pred ;
       | "\exists" binders ";" pred ;
       | "\old" "(" pred ")" | "\at" "(" pred "," label ")" ;
\end{syntax}
