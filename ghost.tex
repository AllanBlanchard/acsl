\begin{syntax}
  
  
  vardecl ::= "//@" "ghost" vardecl 
          | "//@"  \\
   
  typedecl ::= "//@" "ghost" typedecl \\

  statement ::= C-statement | ghost-statements C-statement \\
  ghost-statements ::= "/*@" "ghost" C-statement+ "*/"

\end{syntax}
