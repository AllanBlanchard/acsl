\begin{syntax}
  abrupt-clause ::= throws-clause
  \
  abrupt-clause-stmt ::= throws-clause
  \
  throws-clause ::= "throws" pred ";" ;
                  | "throws" "{" "..." "}" pred ";" ;
                  | "throws" "{" C-type-name ;
                    ( "," C-type-name )* "}" pred ";" ;
  \
  term ::= "\exception"
\end{syntax}
