\begin{example}
  The declaration
  \begin{c}
    //@ logic int f(int x) = x+1 ;
  \end{c}
  is not allowed because \verb!x+1!, which is a mathematical integer, must be
  casted to \texttt{int}.  One should write either
  \begin{c}
    //@ logic integer f(int x) = x+1 ;
  \end{c}
  or
  \begin{c}
    //@ logic int f(int x) = (int)(x+1) ;
  \end{c}
\end{example}
