\begin{syntax}
  term ::=  "[|" "|]" ; empty list
       | "[|" term ("," term)* "|]" ; list of elements
       | term "^" term ; list concatenation (overloading bitwise-xor 
                         ; operator)
       | term "*^" term ; list repetition
\end{syntax}
