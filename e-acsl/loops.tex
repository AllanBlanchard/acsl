\begin{syntax}
  statement ::= "/*@" loop-annot "*/" ;
  "while" "(" expr ")" statement ;
  | "/*@" loop-annot "*/" ;
  "for";
  "(" expr ";" expr ";" expr ")";
  statement ;
  | "/*@" loop-annot "*/" ;
  "do" statement ;
  "while" "(" expr ")" ";"
  \
  loop-annot ::= loop-variant?
  \
  loop-variant ::= "loop" "variant" term ";" ;
  | "loop" "variant" term "for" id ";" ; variant for relation $id$
\end{syntax}
