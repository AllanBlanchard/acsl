\begin{syntax}
  rel-op ::= "==" | "!=" | "<=" | ">=" | ">" | "<"
       \
  pred ::= "\true" | "\false" ;
       | term (rel-op term)+ ; comparisons
       | id "(" term ("," term)* ")" ; predicate application
       | "(" pred ")" ; parentheses
       | [ pred "&&" pred ] ; conjunction
       | [ pred "||" pred ] ; disjunction
       | [ pred "==>" pred ] ; implication
       | pred "<==>" pred ; equivalence
       | "!" pred ; negation
       | [ pred "^^" pred ] ; exclusive or
       | [ term "?" pred ":" pred ] ; ternary condition
       | [ pred "?" pred ":" pred ];
       | "\let" id "=" term ";" pred ; local binding
       | "\let" id "=" pred ";" pred ;
       | [ "\forall" binders ";" ] ;
         [ guards "==>" pred ]; universal guarded quantification
       | [ "\exists" binders ";" ];
         [ guards "&&" pred ] ; existential guarded quantification
       | [ "\forall" boolean-binders ";" pred ]; universal boolean quantification
       | [ "\exists" boolean-binders ";" pred ]; existential boolean quantification
       | id ":" pred ; syntactic naming
       \
  [ guards ] ::= [ interv  ("&&" interv)* ]
       \
  [ interv ]  ::= [ (term guard-op)+ ];
                  [ id ];
                  [ (guard-op term)+ ]
       \
  [ guard-op ] ::= [ "==" | "<=" | "<" ]
\end{syntax}
