\begin{syntax}
  tset ::= term | tset "->" id | tset "." id  ;
       | "*" tset | tset "[" tset "]" | term? ".." term? ; 
       | tset "," tset | tset "+" tset
       | "(" tset ")"
       | "{" tset "|" binders (";" predicate)? "}"
\end{syntax}
