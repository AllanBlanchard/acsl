\begin{syntax}
term ::= "this" ;
  | "\\this" ;
  | term ( "." | "->" ) id "(" ( term ( "," term )* )? ")" ;
\
ident ::= ( qualifier "::" )* qualifier ;
\
poly-id ::= ( qualifier "::" )+ poly-id ;
\
qualifier ::= id ( "<" type-expr ( "," type-expr )* ">" )? ;
\
literal ::= "nullptr" | "true" | "false" ;
\end{syntax}
